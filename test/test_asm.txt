.alu_test:
            LDI 1024
            ADD         ; AC = 1024
            SHR         ; AC = 512
            AND         ; AC = 0
            XOR         ; AC = 1024
            SHR         ; AC = 512
            ORA         ; AC = 1536
            SUB         ; AC = 512
            NOP         ; AC = 512
            MVX         ; IX = 1024
            STA         ; m[1024] = 0, m[1025] = 4
            LDI 0
            MVA         ; AC = 0
            LDW         ; DR = 1024
            MVA         ; AC = 1024
.jmp_test:

            NOP
            NOP
            NOP

            JMP
